## Usage

```tsx
import { OrderEntry } from "@hanzo/ui/finance"

export default function Demo() {
  const handleOrderSubmit = (order) => {
    console.log('Order submitted:', order)
  }

  return (
    <OrderEntry
      symbol="AAPL"
      currentPrice={175.50}
      onSubmit={handleOrderSubmit}
    />
  )
}
```

## Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| symbol | string | - | Trading symbol (required) |
| currentPrice | number | - | Current market price (required) |
| onSubmit | function | - | Callback when order submitted |
| defaultSide | "buy" \| "sell" | "buy" | Default order side |
| defaultOrderType | "market" \| "limit" \| "stop" | "market" | Default order type |
