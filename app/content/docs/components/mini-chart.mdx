## Usage

```tsx
import { MiniChart } from "@hanzo/ui/finance"

export default function Demo() {
  return (
    <MiniChart
      symbol="NASDAQ:AAPL"
      colorTheme="dark"
    />
  )
}
```

## Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| symbol | string | "NASDAQ:AAPL" | Trading symbol to display |
| width | string \| number | "100%" | Chart width |
| height | string \| number | 350 | Chart height |
| colorTheme | "light" \| "dark" | "dark" | Color theme |
| locale | string | "en" | Language locale |
| isTransparent | boolean | false | Transparent background |
| autosize | boolean | true | Auto-resize to container |
| trendLineColor | string | "rgba(41, 98, 255, 1)" | Trend line color |

## Examples

### Basic Usage

```tsx
<MiniChart symbol="NASDAQ:AAPL" />
```

### Custom Styling

```tsx
<MiniChart
  symbol="BINANCE:BTCUSDT"
  colorTheme="light"
  height={250}
  trendLineColor="#00ff00"
/>
```
