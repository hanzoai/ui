## Usage

```tsx
import { SymbolOverview } from "@hanzo/ui/finance"

export default function Demo() {
  return (
    <SymbolOverview
      symbols={[
        ["Apple", "NASDAQ:AAPL"],
        ["Google", "NASDAQ:GOOGL"],
        ["Microsoft", "NASDAQ:MSFT"]
      ]}
      colorTheme="dark"
    />
  )
}
```

## Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| symbols | Array<[string, string]> | [...] | Array of [name, symbol] pairs |
| chartOnly | boolean | false | Show only chart without details |
| width | string \| number | "100%" | Widget width |
| height | string \| number | 400 | Widget height |
| colorTheme | "light" \| "dark" | "dark" | Color theme |
| locale | string | "en" | Language locale |
| isTransparent | boolean | false | Transparent background |
| autosize | boolean | true | Auto-resize to container |
| showVolume | boolean | true | Show volume data |
