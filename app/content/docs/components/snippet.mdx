## Usage

```tsx
import { CodeSnippet } from "@/registry/default/ui/code-snippet"

export default function Example() {
  return (
    <CodeSnippet
      code={`function hello() {\n  console.log("Hello, world!");\n}`}
      language="javascript"
      filename="hello.js"
    />
  )
}
```
